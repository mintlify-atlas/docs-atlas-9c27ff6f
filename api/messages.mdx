---
title: Messages API
description: API reference for managing messages, fragments, and attachments in ZapDev
---

The Messages API handles all message-related operations including creating messages, managing fragments (generated code), and handling attachments.

## Data Model

### Message

<ResponseField name="_id" type="Id<'messages'>" required>
  Unique identifier for the message
</ResponseField>

<ResponseField name="content" type="string" required>
  The text content of the message
</ResponseField>

<ResponseField name="role" type="MessageRole" required>
  Who sent the message: `USER` or `ASSISTANT`
</ResponseField>

<ResponseField name="type" type="MessageType" required>
  Message type: `RESULT`, `ERROR`, or `STREAMING`
</ResponseField>

<ResponseField name="status" type="MessageStatus" required>
  Message status: `PENDING`, `STREAMING`, or `COMPLETE`
</ResponseField>

<ResponseField name="projectId" type="Id<'projects'>" required>
  The project this message belongs to
</ResponseField>

<ResponseField name="createdAt" type="number">
  Timestamp when the message was created (milliseconds)
</ResponseField>

<ResponseField name="updatedAt" type="number">
  Timestamp when the message was last updated (milliseconds)
</ResponseField>

### Fragment

<ResponseField name="_id" type="Id<'fragments'>" required>
  Unique identifier for the fragment
</ResponseField>

<ResponseField name="messageId" type="Id<'messages'>" required>
  The message this fragment belongs to
</ResponseField>

<ResponseField name="sandboxUrl" type="string" required>
  E2B sandbox URL where the code runs
</ResponseField>

<ResponseField name="title" type="string" required>
  Display title for the fragment
</ResponseField>

<ResponseField name="files" type="object" required>
  File structure with paths as keys and content as values
</ResponseField>

<ResponseField name="metadata" type="object">
  Additional metadata about the fragment (optional)
</ResponseField>

<ResponseField name="framework" type="Framework" required>
  Framework used: `NEXTJS`, `ANGULAR`, `REACT`, `VUE`, or `SVELTE`
</ResponseField>

### Attachment

<ResponseField name="_id" type="Id<'attachments'>" required>
  Unique identifier for the attachment
</ResponseField>

<ResponseField name="messageId" type="Id<'messages'>" required>
  The message this attachment belongs to
</ResponseField>

<ResponseField name="type" type="AttachmentType" required>
  Attachment type: `IMAGE`, `FIGMA_FILE`, or `GITHUB_REPO`
</ResponseField>

<ResponseField name="url" type="string" required>
  URL of the attachment
</ResponseField>

<ResponseField name="size" type="number" required>
  File size in bytes
</ResponseField>

<ResponseField name="width" type="number">
  Image width in pixels (optional)
</ResponseField>

<ResponseField name="height" type="number">
  Image height in pixels (optional)
</ResponseField>

<ResponseField name="importId" type="Id<'imports'>">
  Related import ID if from Figma/GitHub (optional)
</ResponseField>

<ResponseField name="sourceMetadata" type="object">
  Metadata from the import source (optional)
</ResponseField>

## Queries

### list

Get all messages for a project with fragments and attachments.

```typescript
import { api } from '@/convex/_generated/api';
import { useQuery } from 'convex/react';

const messages = useQuery(api.messages.list, {
  projectId: "jh7...",
});
```

<ParamField path="projectId" type="Id<'projects'>" required>
  The project ID to get messages for
</ParamField>

**Returns:** Array of messages sorted by creation date (ascending)

<ResponseField name="Fragment" type="object | null">
  The fragment associated with this message (if any)
</ResponseField>

<ResponseField name="Attachment" type="array">
  All attachments for this message
</ResponseField>

**Authentication:** Required. User must own the project.

**Throws:** `"Unauthorized"`

## Database Schema

The `messages` table has the following indexes:

```typescript
// convex/schema.ts
messages: defineTable({
  content: v.string(),
  role: messageRoleEnum,
  type: messageTypeEnum,
  status: messageStatusEnum,
  projectId: v.id("projects"),
  createdAt: v.optional(v.number()),
  updatedAt: v.optional(v.number()),
})
  .index("by_projectId", ["projectId"])
  .index("by_projectId_createdAt", ["projectId", "createdAt"])
```

The `fragments` table:

```typescript
fragments: defineTable({
  messageId: v.id("messages"),
  sandboxUrl: v.string(),
  title: v.string(),
  files: v.any(),
  metadata: v.optional(v.any()),
  framework: frameworkEnum,
  createdAt: v.optional(v.number()),
  updatedAt: v.optional(v.number()),
})
  .index("by_messageId", ["messageId"])
```

The `attachments` table:

```typescript
attachments: defineTable({
  type: attachmentTypeEnum,
  url: v.string(),
  width: v.optional(v.number()),
  height: v.optional(v.number()),
  size: v.number(),
  messageId: v.id("messages"),
  importId: v.optional(v.id("imports")),
  sourceMetadata: v.optional(v.any()),
  createdAt: v.optional(v.number()),
  updatedAt: v.optional(v.number()),
})
  .index("by_messageId", ["messageId"])
```

## Example Usage

### Creating a message in a project

```typescript
import { api } from '@/convex/_generated/api';
import { useAction } from 'convex/react';
import { useState } from 'react';

function MessageForm({ projectId }: { projectId: string }) {
  const [input, setInput] = useState('');
  const createMessage = useAction(api.messages.createWithAttachments);

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    try {
      const result = await createMessage({
        value: input,
        projectId,
      });
      console.log('Created message:', result.messageId);
      setInput('');
    } catch (error) {
      console.error('Failed to create message:', error);
    }
  };

  return (
    <form onSubmit={handleSubmit}>
      <textarea
        value={input}
        onChange={(e) => setInput(e.target.value)}
        placeholder="Describe changes..."
      />
      <button type="submit">Send</button>
    </form>
  );
}
```

### Displaying messages with fragments

```typescript
import { api } from '@/convex/_generated/api';
import { useQuery } from 'convex/react';

function MessageList({ projectId }: { projectId: string }) {
  const messages = useQuery(api.messages.list, { projectId });

  if (!messages) return <div>Loading...</div>;

  return (
    <div>
      {messages.map((message) => (
        <div key={message._id}>
          <p>{message.content}</p>
          {message.Fragment && (
            <div>
              <h4>{message.Fragment.title}</h4>
              <iframe src={message.Fragment.sandboxUrl} />
            </div>
          )}
          {message.Attachment.map((att) => (
            <img key={att._id} src={att.url} alt="Attachment" />
          ))}
        </div>
      ))}
    </div>
  );
}
```
